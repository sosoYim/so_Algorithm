function solution(board){  
    let answer="YES";
    let n=board.length;
    for(let i=0; i<n; i++){
        let ch1 = new Set();
        let ch2 = new Set();
        for(let j=0; j<n; j++){
            ch1.add(board[i][j]);
            ch2.add(board[j][i]);
        }
        if(ch1.size !== 9) return "NO";
        if(ch2.size !== 9) return "NO";
    }
    for(let i=0; i<3; i++){
        for(let j=0; j<3; j++){
            let ch3 = new Set();
            for(let s=0; s<3; s++){
                for(let k=0; k<3; k++){
                    ch3.add(board[i*3+s][j*3+k]);
                }
            }
            if(ch3.size !== 9) return "NO";
        }
    }
    return answer;
}
